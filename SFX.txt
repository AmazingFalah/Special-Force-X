create database SFX;
use SFX;

CREATE TABLE Register (
    Reg_id INT AUTO_INCREMENT PRIMARY KEY,
    Full_Name VARCHAR(100) NOT NULL,
    Username VARCHAR(50) UNIQUE NOT NULL,
    Email VARCHAR(100) UNIQUE NOT NULL,
    Password VARCHAR(255) NOT NULL,
    Date_Of_Birth DATE
);

CREATE TABLE Server (
    SVR_Name VARCHAR(100) NOT NULL UNIQUE,
    SVR_Code VARCHAR(20) NOT NULL,
    SVR_Mode VARCHAR(50) NOT NULL,
    SVR_Description TEXT,
    PRIMARY KEY (SVR_Name)
);

CREATE TABLE Inventory (
    Inv_id INT AUTO_INCREMENT PRIMARY KEY,
    Inv_Name VARCHAR(100) NOT NULL UNIQUE
);

CREATE TABLE Equipment (
    Reg_id INT,
    Inv_id INT,
    PRIMARY KEY (Reg_id),
    FOREIGN KEY (Reg_id) REFERENCES Register(Reg_id) ON DELETE CASCADE,
    FOREIGN KEY (Inv_id) REFERENCES Inventory(Inv_id)
);
